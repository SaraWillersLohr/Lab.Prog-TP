#include <iostream>
#include <vector>
#include <ctime>
#include <cstdlib>

using namespace std;

vector<int> GenerarVector(int N);
int SumaElementosParesDivisiblesPor3(const vector<int>& V);

vector<int> GenerarVector(int N) {
    vector<int> A;
    srand(time(nullptr));
    for (int i = 0; i < N; i++) {
        int e = rand() % 1500;
        A.push_back(e);
    }
    return A;
}

int SumaElementosParesDivisiblesPor3(const vector<int>& V) {
    int suma = 0;
    for (int i = 0; i < V.size(); i++) {
        if (i % 3 == 0 && V[i] % 2 == 0) {
            suma += V[i];
        }
    }
    return suma;
}

int main() {
    int n;
    cout << "Ingrese el tamaÃ±o del vector: ";
    cin >> n;

    vector<int> V = GenerarVector(n);

    cout << "Vector V: ";
    for (int elemento : V) {
        cout << elemento << " ";
    }
    cout << endl;

    int resultado = SumaElementosParesDivisiblesPor3(V);
    cout << "La suma de elementos pares en posiciones divisibles por 3 de V es: " << resultado << endl;

    return 0;
}
